Implementation Reference

CSS Code 
/* Base skeleton styling */
.skeleton {
  background: linear-gradient(90deg, #2f3336 0%, #3f4447 50%, #2f3336 100%);
  background-size: 200% 100%;
  animation: skeleton-pulse 1.5s ease-in-out infinite;
}

@keyframes skeleton-pulse {
  0% { background-position: 200% 0; }
  100% { background-position: -200% 0; }
}

/* Static skeleton (no animation) */
.skeleton-static {
  background: #2f3336;
  animation: none;
}
HTML Usage 
<!-- Text skeleton -->
<div class="skeleton h-4 w-3/4 rounded" aria-hidden="true"></div>

<!-- Circle skeleton (avatar) -->
<div class="skeleton w-12 h-12 rounded-full" aria-hidden="true"></div>

<!-- Rectangle skeleton (image) -->
<div class="skeleton aspect-video w-full rounded-lg" aria-hidden="true"></div>

<!-- Composite: Avatar with text -->
<div class="flex items-center gap-3" aria-hidden="true">
  <div class="skeleton w-10 h-10 rounded-full shrink-0"></div>
  <div class="space-y-2 flex-1">
    <div class="skeleton h-4 w-32 rounded"></div>
    <div class="skeleton h-3 w-24 rounded"></div>
  </div>
</div>
